"bibEntry" -}} {{- if eq $bibEntry.type "book" -}} {{- partial "styles/ieee/types/book" $bibEntry -}} {{- else if (eq $bibEntry.type "chapter") -}} {{- partial "styles/ieee/types/chapter" $bibEntry -}} {{- else if or (eq $bibEntry.type "article-journal") (eq $bibEntry.type "article") -}} {{- partial "styles/ieee/types/article" $bibEntry -}} {{- else if (eq $bibEntry.type "paper-conference") -}} {{- partial "styles/ieee/types/conference-paper" $bibEntry -}} {{- else if eq $bibEntry.type "webpage" -}} {{- partial "styles/ieee/types/webpage" $bibEntry -}} {{- else -}} {{- warnf "Bibliography type '%s' is not implemented. Defaulting to type 'book': %s" $bibEntry.type $context.Position -}} {{- partial "styles/ieee/types/book" $bibEntry -}} {{- end -}}
